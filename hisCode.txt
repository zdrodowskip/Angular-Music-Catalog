musicApp.config(['$routeProvider',
  function($routeProvider) {
    $routeProvider.
      when('/ViewList', {
		templateUrl: 'templates/viewList.html',
		controller: 'listController'
	}).
      when('/ViewDetail', {
		templateUrl: 'templates/viewDetail.html',
		controller: 'detailController'
      }).
      otherwise({
		redirectTo: '/ViewList'
      });
}]);
